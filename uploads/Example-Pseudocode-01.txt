  initialise:
    lambda = generate_random_number

  responses:
    if first round:
      cooperate
    otherwise:
      // Not the first round
      if last opponent response is cooperate:
        // If the opponent last cooperated
        increment lambda
        cooperate
      else:
        // The opponent last defected
        decrement lambda
        defect
